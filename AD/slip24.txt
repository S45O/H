Q 2)

 CREATE
 // Book types
 (text:BookType {name: "Text", characteristics: "Educational, Structured Content"}),
 (ref:BookType {name: "Reference", characteristics: "Informational, Reference Material"}),
 (bio:BookType {name: "Bibliography", characteristics: "List of sources, Citing books"}),
 // Students
 (student1:Student {name: "Amit"}),
 (student2:Student {name: "Priya"}),
 (student3:Student {name: "Rahul"}),
 (student4:Student {name: "Sneha"}),
 // Relationships
 (student1)-[:BUYS]->(text),
 (student1)-[:BUYS]->(ref),
 (student2)-[:BUYS]->(ref),
(student2)-[:BUYS]->(bio),
 (student3)-[:BUYS]->(text),
 (student4)-[:BUYS]->(text),
 (student4)-[:BUYS]->(ref),
 // Recommendations
 (student1)-[:RECOMMENDS {rating: 5}]->(text),
 (student2)-[:RECOMMENDS {rating: 4}]->(ref),
 (student3)-[:RECOMMENDS {rating: 5}]->(bio),
 (student4)-[:RECOMMENDS {rating: 5}]->(text);
 
a. List the books of type “text”.
 MATCH(book:BookType {name: "Text"})
 RETURNbook;
 b. List the names of students who bought text and reference types of books.
 MATCH(student:Student)-[:BUYS]->(book:BookType)
 WHEREbook.name IN ["Text", "Reference"]
 WITH student, COUNT(DISTINCT book) AS bookCount
 WHEREbookCount = 2
 RETURNstudent.name;
 c. List the most recommended book type.
 MATCH(student:Student)-[:RECOMMENDS]->(book:BookType)
 RETURNbook.name, COUNT(student) AS recommendationCount
 ORDERBYrecommendationCount DESC
 LIMIT 1;
 d. List the students who bought more than one type of book.
 MATCH(student:Student)-[:BUYS]->(book:BookType)
 WITH student, COUNT(DISTINCT book) AS bookCount
 WHEREbookCount > 1
 RETURNstudent.name;
