Q 2)

 CREATE (a1:Author {name: "John Doe", genre: "Comics"}),
 (a2:Author {name: "Jane Smith", genre: "Fiction"}),
 (a3:Author {name: "Alex Brown", genre: "Comics"}),
 (p1:Publisher {name: "Sage"}),
 (p2:Publisher {name: "Nature Publishing"}),
 (b1:Book {title: "The Adventures of Tim", genre: "Comics", published_year: 2015}),
 (b2:Book {title: "The Storyteller", genre: "Fiction", published_year: 2018}),
 (r1:Reader {name: "Mike", age: 25}),
 (r2:Reader {name: "Sam", age: 30}),
 (r3:Reader {name: "Lily", age: 22}),
 (a1)-[:WROTE]->(b1),
 (a2)-[:WROTE]->(b2),
 (a3)-[:WROTE]->(b1),
 (p1)-[:PUBLISHED]->(b1),
 (p1)-[:PUBLISHED]->(b2),
 (r1)-[:READ {date_read: "2022-07-01"}]->(b1),
 (r2)-[:RECOMMENDED {date_recommended: "2022-07-15"}]->(b1),
 (r3)-[:REVIEWED {rating: 4, review_text: "Great read"}]->(b1),
 (r1)-[:REVIEWED {rating: 3, review_text: "Nice book"}]->(b2);

 a. List the names of authors who wrote "Comics".
 MATCH(a:Author)-[:WROTE]->(b:Book {genre: "Comics"})
 RETURNa.name;
 b. Count the number of readers of books published by "Sage".
 MATCH(p:Publisher {name: "Sage"})-[:PUBLISHED]->(b:Book)<-[:READ]-(r:Reader)
 RETURNCOUNT(r) AS num_readers;
 c. List all publishers whose name starts with "N".
 MATCH(p:Publisher)
 WHEREp.name STARTS WITH "N"
 RETURNp.name;
 d. List the names of people who have given a rating of 3 or higher for a book.
 MATCH(r:Reader)-[rev:REVIEWED]->(b:Book)
 WHERErev.rating >= 3
 RETURNr.name;
