 (s10:Student {name: "Arjun", category: "OBC", income: 310000, year: 2021}),
 (sch2:Scholarship {name: "OBC Scholarship", category: "OBC", income_limit: 400000, year: 2020}),
 (sch3:Scholarship {name: "SC Scholarship", category: "SC", income_limit: 250000, year: 2020}),
 (sch4:Scholarship {name: "General Scholarship", category: "General", income_limit: 500000, year:
 2021}),
 (sch5:Scholarship {name: "Sports Scholarship", category: "General", income_limit: 550000, year:
 2021}),
 (sch6:Scholarship {name: "Merit Scholarship", category: "General", income_limit: 400000, year: 2021}),
 (s5)-[:APPLIED_FOR {year: 2021}]->(sch2),
 (s4)-[:APPLIED_FOR {year: 2020}]->(sch3),
 (s6)-[:APPLIED_FOR {year: 2020}]->(sch6),
 (s7)-[:APPLIED_FOR {year: 2021}]->(sch2),
 (s8)-[:APPLIED_FOR {year: 2021}]->(sch3),
 (s9)-[:APPLIED_FOR {year: 2020}]->(sch4),
 (s10)-[:APPLIED_FOR {year: 2021}]->(sch2);

 a. List the names of scholarships for the OBC category:
 MATCH(sch:Scholarship {category: "OBC"})
 RETURNsch.name;
 b. Count the number of students who benefited from scholarships in 2020-2021:
 MATCH(s:Student)-[r:APPLIED_FOR]->(sch:Scholarship)
 WHEREr.year IN [2020, 2021]
 RETURNCOUNT(s) AS BenefitedStudents;
 c. Update the income limit for the OBC Scholarship:
 MATCH(sch:Scholarship {name: "OBC Scholarship"})
 SET sch.income_limit = 350000
 RETURNsch;
 d. List the most popular scholarship (applied by the most students):
 MATCH(s:Student)-[r:APPLIED_FOR]->(sch:Scholarship)
 RETURNsch.name, COUNT(s) AS NumberOfApplicants
 ORDERBYNumberOfApplicants DESC
 LIMIT 1;
