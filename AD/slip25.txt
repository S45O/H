 CREATE
 // Departments
 (physics:Department {name: "Physics"}),
 (geography:Department {name: "Geography"}),
 (computer:Department {name: "Computer Science"}),
 (mathematics:Department {name: "Mathematics"}),
 // Courses
 (quantum:Course {name: "Quantum Mechanics", description: "Study of quantum systems"}),
 (thermodynamics:Course {name: "Thermodynamics", description: "Study of heat and energy"}),
 (geography101:Course {name: "Introduction to Geography", description: "Basic concepts in geography"}),
 (dataStructures:Course {name: "Data Structures", description: "Fundamentals of data organization"}),
 (algorithms:Course {name: "Algorithms", description: "Study of algorithm design and analysis"}),
 (calc:Course {name: "Calculus", description: "Study of change and motion"}),
 // Relationships
 (physics)-[:OFFERS]->(quantum),
 (physics)-[:OFFERS]->(thermodynamics),
 (geography)-[:OFFERS]->(geography101),
 (computer)-[:OFFERS]->(dataStructures),
 (computer)-[:OFFERS]->(algorithms),
 (mathematics)-[:OFFERS]->(calc),
(mathematics)-[:OFFERS]->(algorithms),
 // Recommendations (assuming a Person node for recommendations)
 (person1:Person {name: "Alice"}),
 (person2:Person {name: "Bob"}),
 (person1)-[:RECOMMENDS]->(quantum),
 (person1)-[:RECOMMENDS]->(geography101),
 (person2)-[:RECOMMENDS]->(algorithms);
 
a. List the details of all the departments in the university.
 MATCH(dept:Department)
 RETURNdept;
 b. List the names of the courses provided by the Physics department.
 MATCH(physics:Department {name: "Physics"})-[:OFFERS]->(course:Course)
 RETURNcourse.name;
 c. List the most recommended course in the Geography department.
 MATCH(geo:Department {name:
 "Geography"})-[:OFFERS]->(course:Course)<-[:RECOMMENDS]-(person:Person)
 RETURNcourse.name, COUNT(person) AS recommendationCount
 ORDERBYrecommendationCount DESC
 LIMIT 1;
 d. List the names of common courses across Mathematics and Computer Science departments.
 MATCH(math:Department {name:
 "Mathematics"})-[:OFFERS]->(course:Course)<-[:OFFERS]-(comp:Department {name: "Computer
 Science"})
 RETURNcourse.name;
