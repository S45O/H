    function rotateClockwise() {
      angle += 45;
      image.style.transform = rotate(${angle}deg);

      angle -= 45;

CREATE
// Vehicle types
 (tw:VehicleType {name: "Two-Wheeler", characteristics: "Lightweight, Fuel-efficient, Easy to park"}),
 (fw:VehicleType {name: "Four-Wheeler", characteristics: "More space, Comfortable, Safer"}),
 // Customers
 (cust1:Customer {name: "Amit"}),
 (cust2:Customer {name: "Priya"}),
 (cust3:Customer {name: "Rahul"}),
 (cust4:Customer {name: "Sneha"}),
 // Recommendations
 (cust1)-[:BUYS]->(tw),
 (cust2)-[:BUYS]->(fw),
 (cust3)-[:BUYS]->(tw),
 (cust3)-[:BUYS]->(fw),
 (cust4)-[:BUYS]->(tw),
 (cust1)-[:RECOMMENDS {rating: 5}]->(tw),
 (cust2)-[:RECOMMENDS {rating: 4}]->(fw),
 (cust3)-[:RECOMMENDS {rating: 5}]->(tw),
 (cust4)-[:RECOMMENDS {rating: 5}]->(fw);

 a. List the characteristics of four-wheeler types.
 MATCH(fw:VehicleType {name: "Four-Wheeler"})
 RETURNfw.characteristics;
 b. List the names of customers who bought a two-wheeler vehicle.
 MATCH(cust:Customer)-[:BUYS]->(tw:VehicleType {name: "Two-Wheeler"})
 RETURNcust.name;
 c. List the customers who bought more than one type of vehicle.
 MATCH(cust:Customer)-[:BUYS]->(v:VehicleType)
 WITH cust, COUNT(v) AS vehicleCount
 WHEREvehicleCount > 1
 RETURNcust.name;
 d. List the most recommended vehicle type.
 MATCH(cust:Customer)-[:RECOMMENDS]->(v:VehicleType)
 RETURNv.name, COUNT(cust) AS recommendationCount
 ORDERBYrecommendationCount DESC
 LIMIT 1;
