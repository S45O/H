 CREATE
 // Car models
(honda:CarModel {name: "Honda City"}),
 (skoda:CarModel {name: "Skoda"}),
 (creta:CarModel {name: "Creta"}),
 (swift:CarModel {name: "Swift"}),
 (ertiga:CarModel {name: "Ertiga"}),
 // Sections
 (section1:Section {name: "Section 1"}),
 (section2:Section {name: "Section 2"}),
 (section3:Section {name: "Section 3"}),
 // Sales staff
 (narayan:SalesStaff {name: "Mr. Narayan"}),
 (ravi:SalesStaff {name: "Mr. Ravi"}),
 // Customers
 (customer1:Customer {name: "Amit"}),
 (customer2:Customer {name: "Priya"}),
 (customer3:Customer {name: "Rahul"}),
 // Relationships
 (section1)-[:CONTAINS]->(honda),
 (section1)-[:CONTAINS]->(swift),
 (section2)-[:CONTAINS]->(skoda),
 (section3)-[:CONTAINS]->(creta),
 (section3)-[:CONTAINS]->(ertiga),
 (narayan)-[:HANDLES]->(section1),
 (ravi)-[:HANDLES]->(section2),
 (customer1)-[:MAKES]->(enquiry1:Enquiry {query: "Inquiry about Honda City"}),
 (customer2)-[:MAKES]->(enquiry2:Enquiry {query: "Inquiry about Creta"}),
 (customer3)-[:MAKES]->(enquiry3:Enquiry {query: "Inquiry about Skoda"}),
 // Purchases
 (customer1)-[:PURCHASED]->(honda),
 (customer2)-[:PURCHASED]->(creta);
s
 a. List the types of cars available in the showroom.
 MATCH(car:CarModel)
 RETURNcar.name;
 b. List the sections handled by Mr. Narayan.
 MATCH(narayan:SalesStaff {name: "Mr. Narayan"})-[:HANDLES]->(section:Section)
 RETURNsection.name;
 c. List the names of customers who have done only enquiry but not made any purchase.
 MATCH(customer:Customer)
 WHERENOT(customer)-[:PURCHASED]->(:CarModel)
 RETURNcustomer.name;
 d. List the highly sold car model.
 MATCH(customer:Customer)-[:PURCHASED]->(car:CarModel)
 RETURNcar.name, COUNT(customer) AS sales
 ORDERBYsales DESC
 LIMIT 1;
